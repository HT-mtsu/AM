      # Import necessary libraries
      import numpy as np
      from numpy import linalg as LA
      
      # Define parameters
      t = 1.0
      L = 10
      H = np.zeros((L,L))
      np.fill_diagonal(H, -2*t)
      np.fill_diagonal(H[1:], t)
      np.fill_diagonal(H[:,1:], t)
      H[0,-1] = t
      H[-1,0] = t
      
      # Diagonalize the Hamiltonian
      eigvals, eigvecs = LA.eigh(H)
      console.log('Eigenvalues: ' + eigvals);
